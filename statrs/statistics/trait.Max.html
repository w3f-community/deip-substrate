Expand description

The Max trait specifies that an object has a maximum value

Required methods

fn max(&self) -> T

[src]

Returns the maximum value in the domain of a given distribution if it exists, otherwise None.

Examples

use statrs::statistics::Max;
use statrs::distribution::Uniform;

let n = Uniform::new(0.0, 1.0).unwrap();
assert_eq!(1.0, n.max());

impl<D: AsMut<[f64]> + AsRef<[f64]>> Max<f64> for Data<D>

[src]

fn max(&self) -> f64

[src]

Returns the maximum value in the data

Remarks

Returns f64::NAN if data is empty or an entry is f64::NAN

Examples

use statrs::statistics::Max;
use statrs::statistics::Data;

let x = [];
let x = Data::new(x);
assert!(x.max().is_nan());

let y = [0.0, f64::NAN, 3.0, -2.0];
let y = Data::new(y);
assert!(y.max().is_nan());

let z = [0.0, 3.0, -2.0];
let z = Data::new(z);
assert_eq!(z.max(), 3.0);
